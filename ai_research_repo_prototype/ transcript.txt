import streamlit as st
from pathlib import Path

# --- UI Setup ---
st.set_page_config(page_title="Lab Workflow Transcript Viewer", layout="wide")
st.title("📄 Lab Research Transcript Viewer")
st.caption("Use this to confirm that your file is loaded and input box works")

# --- Load transcript.txt from file ---
file_path = Path("transcript.txt")
if file_path.exists():
    transcript = file_path.read_text(encoding="utf-8")
    st.success("✅ transcript.txt loaded successfully.")
else:
    st.error("❌ transcript.txt not found in the current folder.")
    st.stop()

# --- Show preview of the transcript ---
with st.expander("🔍 View transcript preview"):
    st.text(transcript[:5000])  # Show first 5000 characters

# --- Input box for simulated query ---
query = st.text_input("❓ Type a question you'd like to ask based on the transcript")

if query:
    st.markdown(f"**💬 You asked:** {query}")
    st.info("🧠 This is a placeholder — no AI response yet. We'll add retrieval + answer next.")
